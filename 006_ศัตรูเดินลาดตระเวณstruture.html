Love2D Enemy Patrol Code

local function newEnemy(x, y, step)
    local enemy = {
        x = x,
        y = y,
        speed = 50,
        step = step,
        direction = 1, -- 1 for down, -1 for up
        image = love.graphics.newImage("assets/enemy.png"),
        
        update = function(self, dt)
            self.y = self.y + self.speed * dt * self.direction
            -- Change direction when reaching the step
            if self.y >= self.step + y then
                self.direction = -1
            elseif self.y <= y then
                self.direction = 1
            end
        end,
        
        draw = function(self)
            love.graphics.draw(self.image, self.x, self.y)
        end
    }
    return enemy
end

return newEnemy
